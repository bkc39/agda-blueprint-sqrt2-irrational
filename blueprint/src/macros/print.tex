% Make sure the arguments of \uses and \proves are real labels by emitting
% invisible refs: LaTeX warns about an undefined label but nothing is printed.
% This uses LaTeX3 programming, hence the expl3 syntax.
\ExplSyntaxOn
\NewDocumentCommand{\uses}{m}
 {\clist_map_inline:nn{#1}{\vphantom{\ref{##1}}}%
  \ignorespaces}
\NewDocumentCommand{\proves}{m}
 {\clist_map_inline:nn{#1}{\vphantom{\ref{##1}}}%
  \ignorespaces}
\ExplSyntaxOff
